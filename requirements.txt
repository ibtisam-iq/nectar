# MkDocs static site generator - transforms Markdown into HTML documentation
mkdocs>=1.6.1,<2.0.0

# -----------------------------------------------------------------------------

# -----------------------------------------------------------------------------

# Material theme extension utilities
# Provides additional components and enhancements for Material theme
mkdocs-material-extensions>=1.3.1

# PyMdown Extensions - Advanced Markdown features

pymdown-extensions==10.16.1

# -----------------------------------------------------------------------------
# NAVIGATION & STRUCTURE PLUGINS
# -----------------------------------------------------------------------------
# Awesome Pages - Simplifies navigation with .pages files in folders
# Allows automatic nav generation and custom ordering per directory
mkdocs-awesome-pages-plugin==2.10.1

# Section Index - Creates index pages for navigation sections
# Allows README.md files to serve as section landing pages
mkdocs-section-index

# Same Dir - Enables using repo root as docs directory (docs_dir: .)
# Allows keeping markdown files alongside code instead of in docs/ folder
mkdocs-same-dir

# -----------------------------------------------------------------------------
# LINKING & REFERENCES
# -----------------------------------------------------------------------------
# AutoRefs - Automatic cross-reference linking between pages
# Enables simplified linking syntax and validates internal links
mkdocs-autorefs==1.4.3

# -----------------------------------------------------------------------------
# GIT INTEGRATION
# -----------------------------------------------------------------------------
# Git Revision Date Localized - Shows last updated date from git commits
# Automatically adds "Last updated: <date>" footer to every page
# Uses git history to track when each file was last modified
mkdocs-git-revision-date-localized-plugin==1.4.7

# -----------------------------------------------------------------------------
# OPTIMIZATION & PERFORMANCE
# -----------------------------------------------------------------------------
# Minify - Compresses HTML/CSS/JS for faster page loads
# Removes whitespace, comments, and unnecessary characters from output
# Critical for production deployments (reduces bandwidth and improves SEO)
mkdocs-minify-plugin

# =============================================================================
# INSTALLATION INSTRUCTIONS
# =============================================================================
#
# Standard installation:
#   pip install -r requirements.txt
#
# Install with social cards support:
#   1. Uncomment pillow and cairosvg above
#   2. On macOS: brew install cairo pango gdk-pixbuf libffi
#   3. On Ubuntu: apt-get install libcairo2-dev libpango1.0-dev libgdk-pixbuf2.0-dev
#   4. pip install -r requirements.txt
#
# Install in virtual environment (recommended):
#   python -m venv .venv
#   source .venv/bin/activate  # On Windows: .venv\Scripts\activate
#   pip install -r requirements.txt
